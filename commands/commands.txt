### Backup

# rsync
--dry-run = test run
sudo rsync -aAXv --delete (--dry-run) --exclude=/home/martin/.cache/* --exclude=/media/* --exclude=/mnt/* --exclude=/run/* --exclude=/dev/* --exclude=/dev/* --exclude=/proc/* --exclude=/sys/* --exclude=".cache" --exclude="lost+found" --exclude=/tmp/* / /media/NAS/Martin/Backup/
#restore
1. boot from live usb
2. mount NAS
3. mkdir /mnt/system
4. mkdir /mnt/usb
5. lsblk (show names of devices)
6. sda1 = system (10G size)
7. sdb1 = usb
8. mount /dev/sda1 /mnt/system/ mount system to the system folder
9. mount /dev/sdb1 /mnt/usb mount usb to usb folder
restore
10. rsync -aAXc --delete --exclude="lost+found" /mnt/usb/ /mnt/system (kein /!)
